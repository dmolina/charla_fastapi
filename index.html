Javascript

app = new Vue({
    el: '#app',
    created() {
        this.fetchData();
    },
    data: {
        categories: []
    },
    methods: {
        fetchData() {
            axios.get('http://localhost:8000/categories').then(response => {
                this.categories = response.data.categories;
            });
        }
    }
});
